  \textbf{基本类型符号(ground type symbol)} $\textbf{A},\textbf{B},\textbf{C},\cdots$
  \item 可以为空的一组
  \textbf{函数符号(function symbol)} $\textbf{f},\textbf{g},\textbf{h},\cdots$
\end{itemize}
